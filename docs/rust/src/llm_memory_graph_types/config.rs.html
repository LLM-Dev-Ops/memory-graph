1//! Configuration for the memory graph
2
3use std::path::PathBuf;
4
5/// Configuration for `MemoryGraph`
6#[derive(Debug, Clone)]
7pub struct Config {
8    /// Path to the database directory
9    pub path: PathBuf,
10    /// Cache size in megabytes
11    pub cache_size_mb: usize,
12    /// Enable write-ahead logging for durability
13    pub enable_wal: bool,
14    /// Compression level (0-9, 0 = no compression)
15    pub compression_level: u8,
16    /// Flush interval in milliseconds (0 = sync every write)
17    pub flush_interval_ms: u64,
18}
19
20impl Config {
21    /// Create a new configuration with custom path
22    #[must_use]
23    pub fn new<P: Into<PathBuf>>(path: P) -> Self {
24        Self {
25            path: path.into(),
26            cache_size_mb: 100,
27            enable_wal: true,
28            compression_level: 3,
29            flush_interval_ms: 1000,
30        }
31    }
32
33    /// Set the cache size
34    #[must_use]
35    pub fn with_cache_size(mut self, size_mb: usize) -> Self {
36        self.cache_size_mb = size_mb;
37        self
38    }
39
40    /// Enable or disable write-ahead logging
41    #[must_use]
42    pub const fn with_wal(mut self, enable: bool) -> Self {
43        self.enable_wal = enable;
44        self
45    }
46
47    /// Set compression level (0-9)
48    #[must_use]
49    pub fn with_compression(mut self, level: u8) -> Self {
50        self.compression_level = level.min(9);
51        self
52    }
53
54    /// Set flush interval in milliseconds
55    #[must_use]
56    pub const fn with_flush_interval(mut self, interval_ms: u64) -> Self {
57        self.flush_interval_ms = interval_ms;
58        self
59    }
60}
61
62impl Default for Config {
63    fn default() -> Self {
64        Self {
65            path: PathBuf::from("./data/graph.db"),
66            cache_size_mb: 100,
67            enable_wal: true,
68            compression_level: 3,
69            flush_interval_ms: 1000,
70        }
71    }
72}

74#[cfg(test)]
75mod tests {
76    use super::*;
77
78    #[test]
79    fn test_default_config() {
80        let config = Config::default();
81        assert_eq!(config.cache_size_mb, 100);
82        assert!(config.enable_wal);
83    }
84
85    #[test]
86    fn test_builder_pattern() {
87        let config = Config::new("./test.db")
88            .with_cache_size(200)
89            .with_wal(false)
90            .with_compression(5)
91            .with_flush_interval(2000);
92
93        assert_eq!(config.cache_size_mb, 200);
94        assert!(!config.enable_wal);
95        assert_eq!(config.compression_level, 5);
96        assert_eq!(config.flush_interval_ms, 2000);
97    }
98
99    #[test]
100    fn test_compression_clamping() {
101        let config = Config::default().with_compression(15);
102        assert_eq!(config.compression_level, 9);
103    }
