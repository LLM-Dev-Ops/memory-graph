11/// Types of edges that can connect nodes
12#[derive(Debug, Clone, PartialEq, Eq, Hash, Serialize, Deserialize)]
13pub enum EdgeType {
14    /// Connects sequential prompts in a conversation (Prompt → Prompt)
15    Follows,
16    /// Links a response to its originating prompt (Response → Prompt)
17    RespondsTo,
18    /// Tracks which agent handled a prompt (Prompt → Agent)
19    HandledBy,
20    /// Links a prompt to the session it belongs to (Prompt → Session)
21    PartOf,
22    /// Links a response to the tools it invoked (Response → ToolInvocation)
23    Invokes,
24    /// Links a response to the agent it handed off to (Response → Agent)
25    TransfersTo,
26    /// Links a prompt to the template it was created from (Prompt → Template)
27    Instantiates,
28    /// Links a template to its parent template (Template → Template)
29    Inherits,
30    /// Links a prompt to external context sources (Prompt → ExternalContext)
31    References,
32}

36/// Properties for INSTANTIATES edge (Prompt → Template)
37///
38/// Tracks how a prompt was generated from a template, including the template
39/// version and variable bindings used during instantiation.
40#[derive(Debug, Clone, Serialize, Deserialize)]
41pub struct InstantiatesProperties {
42    /// Version of the template that was used
43    pub template_version: String,
44    /// Variable bindings used during instantiation
45    pub variable_bindings: HashMap<String, String>,
46    /// When the template was instantiated
47    pub instantiation_time: DateTime<Utc>,
48}
49
50impl InstantiatesProperties {
51    /// Create new instantiation properties
52    pub fn new(template_version: String, variable_bindings: HashMap<String, String>) -> Self {
53        Self {
54            template_version,
55            variable_bindings,
56            instantiation_time: Utc::now(),
57        }
58    }
59
60    /// Convert to property map for storage
61    pub fn to_properties(&self) -> HashMap<String, String> {
62        let mut props = HashMap::new();
63        props.insert(
64            "template_version".to_string(),
65            self.template_version.clone(),
66        );
67        props.insert(
68            "variable_bindings".to_string(),
69            serde_json::to_string(&self.variable_bindings).unwrap_or_default(),
70        );
71        props.insert(
72            "instantiation_time".to_string(),
73            self.instantiation_time.to_rfc3339(),
74        );
75        props
76    }
77
78    /// Parse from property map
79    pub fn from_properties(props: &HashMap<String, String>) -> Result<Self, String> {
80        let template_version = props
81            .get("template_version")
82            .ok_or("Missing template_version")?
83            .clone();
84
85        let variable_bindings = props
86            .get("variable_bindings")
87            .and_then(|s| serde_json::from_str(s).ok())
88            .unwrap_or_default();
89
90        let instantiation_time = props
91            .get("instantiation_time")
92            .and_then(|s| DateTime::parse_from_rfc3339(s).ok())
93            .map(|dt| dt.with_timezone(&Utc))
94            .unwrap_or_else(Utc::now);
95
96        Ok(Self {
97            template_version,
98            variable_bindings,
99            instantiation_time,
100        })
101    }
102}

165            override_sections,
166            version_diff,
167            inheritance_depth,
168        })
169    }

233/// Priority level for agent handoffs
234#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
235pub enum Priority {
236    /// Low priority - can be handled later
237    Low,
238    /// Normal priority - standard handling
239    Normal,
240    /// High priority - expedited handling
241    High,
242    /// Critical priority - immediate attention required
243    Critical,
244}
245
246impl std::fmt::Display for Priority {
247    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
248        match self {
249            Priority::Low => write!(f, "low"),
250            Priority::Normal => write!(f, "normal"),
251            Priority::High => write!(f, "high"),
252            Priority::Critical => write!(f, "critical"),
253        }
254    }
255}
256
257impl std::str::FromStr for Priority {
258    type Err = String;
259
260    fn from_str(s: &str) -> Result<Self, Self::Err> {
261        match s.to_lowercase().as_str() {
262            "low" => Ok(Priority::Low),
263            "normal" => Ok(Priority::Normal),
264            "high" => Ok(Priority::High),
265            "critical" => Ok(Priority::Critical),
266            _ => Err(format!("Invalid priority: {}", s)),
267        }
268    }
269}

371/// Properties for REFERENCES edge (Prompt → ExternalContext)
372///
373/// Tracks external context sources used in prompt generation, including
374/// relevance scores and specific content references.
375#[derive(Debug, Clone, Serialize, Deserialize)]
376pub struct ReferencesProperties {
377    /// Type of external context
378    pub context_type: ContextType,
379    /// Relevance score (0.0 to 1.0)
380    pub relevance_score: f32,
381    /// Optional identifier for specific chunk or section
382    pub chunk_id: Option<String>,
383}
384
385impl ReferencesProperties {
386    /// Create new reference properties
387    pub fn new(context_type: ContextType, relevance_score: f32, chunk_id: Option<String>) -> Self {
388        // Clamp relevance score between 0.0 and 1.0
389        let relevance_score = relevance_score.clamp(0.0, 1.0);
390
391        Self {
392            context_type,
393            relevance_score,
394            chunk_id,
395        }
396    }
397
398    /// Convert to property map for storage
399    pub fn to_properties(&self) -> HashMap<String, String> {
400        let mut props = HashMap::new();
401        props.insert("context_type".to_string(), self.context_type.to_string());
402        props.insert(
403            "relevance_score".to_string(),
404            self.relevance_score.to_string(),
405        );
406        if let Some(ref chunk_id) = self.chunk_id {
407            props.insert("chunk_id".to_string(), chunk_id.clone());
408        }
409        props
410    }
411
412    /// Parse from property map
413    pub fn from_properties(props: &HashMap<String, String>) -> Result<Self, String> {
414        let context_type = props
415            .get("context_type")
416            .and_then(|s| s.parse().ok())
417            .ok_or("Missing or invalid context_type")?;
418
419        let relevance_score = props
420            .get("relevance_score")
421            .and_then(|s| s.parse::<f32>().ok())
422            .unwrap_or(0.0_f32)
423            .clamp(0.0, 1.0);
424
425        let chunk_id = props.get("chunk_id").cloned();
426
427        Ok(Self {
428            context_type,
429            relevance_score,
430            chunk_id,
431        })
432    }
433}

435/// An edge connecting two nodes in the graph
436#[derive(Debug, Clone, Serialize, Deserialize)]
437pub struct Edge {
438    /// Unique edge identifier
439    pub id: EdgeId,
440    /// Source node ID
441    pub from: NodeId,
442    /// Target node ID
443    pub to: NodeId,
444    /// Type of relationship
445    pub edge_type: EdgeType,
446    /// When the edge was created
447    pub created_at: DateTime<Utc>,
448    /// Additional properties for this edge
449    pub properties: HashMap<String, String>,
450}
451
452impl Edge {
453    /// Create a new edge between two nodes
454    #[must_use]
455    pub fn new(from: NodeId, to: NodeId, edge_type: EdgeType) -> Self {
456        Self {
457            id: EdgeId::new(),
458            from,
459            to,
460            edge_type,
461            created_at: Utc::now(),
462            properties: HashMap::new(),
463        }
464    }
465
466    /// Create an edge with custom properties
467    #[must_use]
468    pub fn with_properties(
469        from: NodeId,
470        to: NodeId,
471        edge_type: EdgeType,
472        properties: HashMap<String, String>,
473    ) -> Self {
474        Self {
475            id: EdgeId::new(),
476            from,
477            to,
478            edge_type,
479            created_at: Utc::now(),
480            properties,
481        }
482    }
483
484    /// Add a property to the edge
485    pub fn add_property(&mut self, key: String, value: String) {
486        self.properties.insert(key, value);
487    }
488
489    /// Get a property value
490    #[must_use]
491    pub fn get_property(&self, key: &str) -> Option<&String> {
492        self.properties.get(key)
493    }
