1//! Client implementation for the LLM Memory Graph service
2
3use crate::error::{ClientError, Result};
4use std::collections::HashMap;
5use tonic::transport::Channel;
6
7/// Generated protobuf definitions for the Memory Graph service
8#[allow(missing_docs)]
9pub mod proto {
10    tonic::include_proto!("llm.memory.graph.v1");
11}
12
13use proto::memory_graph_service_client::MemoryGraphServiceClient;
14
15/// High-level client for the LLM Memory Graph service
16#[derive(Clone)]
17pub struct MemoryGraphClient {
18    client: MemoryGraphServiceClient<Channel>,
19}
20
21impl MemoryGraphClient {
22    /// Connect to the Memory Graph service
23    pub async fn connect<D>(addr: D) -> Result<Self>
24    where
25        D: TryInto<tonic::transport::Endpoint>,
26        D::Error: std::error::Error + Send + Sync + 'static,
27    {
28        let endpoint = addr
29            .try_into()
30            .map_err(|e| ClientError::Connection(format!("{:?}", e)))?;
31
32        let client = MemoryGraphServiceClient::connect(endpoint).await?;
33
34        Ok(Self { client })
35    }
36
37    /// Create a new session
38    pub async fn create_session(&self, metadata: HashMap<String, String>) -> Result<String> {
39        let request = proto::CreateSessionRequest { metadata };
40        let response = self.client.clone().create_session(request).await?;
41        Ok(response.into_inner().id)
42    }
43
44    /// Get a session by ID
45    pub async fn get_session(&self, session_id: String) -> Result<proto::Session> {
46        let request = proto::GetSessionRequest {
47            session_id,
48        };
49        let response = self.client.clone().get_session(request).await?;
50        Ok(response.into_inner())
51    }
52
53    /// Delete a session
54    pub async fn delete_session(&self, session_id: String) -> Result<()> {
55        let request = proto::DeleteSessionRequest { session_id };
56        self.client.clone().delete_session(request).await?;
57        Ok(())
58    }
59
60    /// List sessions
61    pub async fn list_sessions(&self, limit: i32, offset: i32) -> Result<Vec<proto::Session>> {
62        let request = proto::ListSessionsRequest { limit, offset };
63        let response = self.client.clone().list_sessions(request).await?;
64        Ok(response.into_inner().sessions)
65    }
66
67    /// Add a prompt
68    pub async fn add_prompt(
69        &self,
70        session_id: String,
71        content: String,

73    ) -> Result<proto::PromptNode> {
74        let request = proto::AddPromptRequest {
75            session_id,
76            content,
77            metadata,
78        };
79        let response = self.client.clone().add_prompt(request).await?;
80        Ok(response.into_inner())
81    }
82
83    /// Add a response
84    pub async fn add_response(
85        &self,
86        prompt_id: String,
87        content: String,
88        token_usage: Option<proto::TokenUsage>,
89        metadata: Option<proto::ResponseMetadata>,
90    ) -> Result<proto::ResponseNode> {
91        let request = proto::AddResponseRequest {
92            prompt_id,
93            content,
94            token_usage,
95            metadata,
96        };
97        let response = self.client.clone().add_response(request).await?;
98        Ok(response.into_inner())
99    }
100
101    /// Query nodes
102    pub async fn query(
103        &self,
104        session_id: Option<String>,
105        node_type: Option<i32>,
106        limit: i32,
107        offset: i32,
108    ) -> Result<proto::QueryResponse> {
109        let request = proto::QueryRequest {
110            session_id,
111            node_type,
112            limit,
113            offset,
114            filters: HashMap::new(),
115            after: None,
116            before: None,
117        };
118        let response = self.client.clone().query(request).await?;
119        Ok(response.into_inner())
120    }
121
122    /// Get service health
123    pub async fn health(&self) -> Result<proto::HealthResponse> {
124        let request = tonic::Request::new(());

133        Ok(response.into_inner())
134    }
